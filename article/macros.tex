%% Ideal functionality, and protocol environments
\newcounter{funcc}
\newcounter{protc}
\renewcommand{\thefuncc}{\arabic{funcc}}
\renewcommand{\theprotc}{\arabic{protc}}

\newenvironment{functionality}[2]{%
\refstepcounter{funcc}%
\label{#1}%
\begin{framed}
        \noindent\scriptsize\textbf{Functionality~\thefuncc} #2
        \\
}%
{
        \end{framed}
}

\newenvironment{protocol}[2]{%
\refstepcounter{protc}%
\label{#1}%
\begin{framed}
        \noindent\scriptsize\textbf{Protocol~\theprotc} #2
        \\
}%
{
        \end{framed}
}
